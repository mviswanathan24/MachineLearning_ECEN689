def transform_X(X):
    for i in range(len(X[:,1])):
        if(i==5):
            plt.imshow(X[i,:].astype("uint8"))
            plt.show()
        temp_2=np.reshape(X[i,:], (X[i,:].shape[0] * X[i,:].shape[1]* X[i,:].shape[2]))        
        temp_3=temp_2[permute]
        X[i,:]=np.reshape(temp_3, X[i,:].shape)        
        if(i==500):
            plt.imshow(X[i,:].astype("uint8"))
            plt.show()
    return X
